Процедура ЗагрузитьНаСервере()
	Excel = Новый COMОбъект("Excel.Application");
	Excel.WorkBooks.Open(ПутьКФайлу);
	Excel.Sheets(1).Select();
	Index = 1;
	
	Пока ЗначениеЗаполнено(Excel.Cells(Index, 1).Value) Цикл
		Об  = Справочники.Сотрудники.СоздатьЭлемент();
		Об.Наименование = Excel.Cells(Index,1).Value;
		Об.НомерТелефона = Excel.Cells(Index,2).Value;
		Об.Должность = Справочники.Должности.НайтиПоНаименованию(Excel.Cells(Index,3).Value);
		Index = Index + 1;
		Об.Записать();
	КонецЦикла;     
	Excel.ActiveWorkBook.Close();
